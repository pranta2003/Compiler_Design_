%{
#include <stdio.h>
#include <stdlib.h>

int yylex();
void yyerror(const char *s);
%}

%token NUMBER

%left '+' '-'
%left '*' '/'
%right UMINUS

%%

input:
      expr '\n'    { printf("Result = %d\n", $1); }
    ;

expr:
      expr '+' expr        { $$ = $1 + $3; }
    | expr '-' expr        { $$ = $1 - $3; }
    | expr '*' expr        { $$ = $1 * $3; }
    | expr '/' expr        {
                               if ($3 == 0) {
                                   printf("Error: Division by zero\n");
                                   exit(1);
                               }
                               $$ = $1 / $3;
                           }
    | '-' expr  %prec UMINUS { $$ = -$2; }
    | '(' expr ')'          { $$ = $2; }
    | NUMBER
    ;

%%

void yyerror(const char *s) {
    printf("Invalid expression: %s\n", s);
}

int main() {
    printf("Enter an expression: ");
    yyparse();
    return 0;
}
